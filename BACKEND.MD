## 2. Configuration (SQLite → PostgreSQL Switch)

```python
# config.py
import os
from dotenv import load_dotenv
load_dotenv()

class Config:
    SECRET_KEY = os.getenv("SECRET_KEY", "dev-secret-key")
    SQLALCHEMY_TRACK_MODIFICATIONS = False

class DevelopmentConfig(Config):
    # ── SQLite3 (used during development) ──
    SQLALCHEMY_DATABASE_URI = "sqlite:///eventflow.db"
    DEBUG = True

class ProductionConfig(Config):
    # ── PostgreSQL (switch for deployment) ──
    # Set DATABASE_URL in .env as:
    # DATABASE_URL=postgresql://user:password@host:5432/eventflow_db
    SQLALCHEMY_DATABASE_URI = os.getenv("DATABASE_URL")
    DEBUG = False

config_map = {
    "development": DevelopmentConfig,
    "production":  ProductionConfig,
}
```

## 3. App Factory

```python
# app/__init__.py
from flask import Flask
from flask_sqlalchemy import SQLAlchemy
from flask_migrate import Migrate
from flask_cors import CORS
import os

db = SQLAlchemy()
migrate = Migrate()

def create_app():
    app = Flask(__name__)

    env = os.getenv("FLASK_ENV", "development")
    from config import config_map
    app.config.from_object(config_map[env])

    db.init_app(app)
    migrate.init_app(app, db)
    CORS(app, origins=["http://localhost:5173"])  # Vite dev server

    # Register blueprints
    from app.routes.auth     import auth_bp
    from app.routes.events   import events_bp
    from app.routes.bookings import bookings_bp
    from app.routes.admin    import admin_bp

    app.register_blueprint(auth_bp,     url_prefix="/api/auth")
    app.register_blueprint(events_bp,   url_prefix="/api/events")
    app.register_blueprint(bookings_bp, url_prefix="/api/bookings")
    app.register_blueprint(admin_bp,    url_prefix="/api/admin")

    return app
```

## 5. Auth Middleware (JWT)

```python
# app/middleware/auth_middleware.py
import jwt
import os
from functools import wraps
from flask import request, jsonify
from app.models.user import User

def token_required(f):
    @wraps(f)
    def decorated(*args, **kwargs):
        token = request.headers.get("Authorization", "").replace("Bearer ", "")
        if not token:
            return jsonify({"error": "Token missing"}), 401
        try:
            data = jwt.decode(token, os.getenv("SECRET_KEY"), algorithms=["HS256"])
            current_user = User.query.get(data["user_id"])
        except jwt.ExpiredSignatureError:
            return jsonify({"error": "Token expired"}), 401
        except Exception:
            return jsonify({"error": "Invalid token"}), 401
        return f(current_user, *args, **kwargs)
    return decorated

def role_required(*roles):
    def decorator(f):
        @wraps(f)
        @token_required
        def decorated(current_user, *args, **kwargs):
            if current_user.role not in roles:
                return jsonify({"error": "Access denied"}), 403
            return f(current_user, *args, **kwargs)
        return decorated
    return decorator
```

## 6. API Routes

### Auth Routes `/api/auth`
```python
# app/routes/auth.py
from flask import Blueprint, request, jsonify
import bcrypt, jwt, os
from datetime import datetime, timedelta
from app import db
from app.models.user import User

auth_bp = Blueprint("auth", __name__)

@auth_bp.route("/register", methods=["POST"])
def register():
    data = request.get_json()
    if User.query.filter_by(email=data["email"]).first():
        return jsonify({"error": "Email already registered"}), 409
    hashed = bcrypt.hashpw(data["password"].encode(), bcrypt.gensalt())
    user = User(name=data["name"], email=data["email"],
                password=hashed.decode(), role=data.get("role", "user"))
    db.session.add(user)
    db.session.commit()
    return jsonify({"message": "Registered successfully", "user": user.to_dict()}), 201

@auth_bp.route("/login", methods=["POST"])
def login():
    data = request.get_json()
    user = User.query.filter_by(email=data["email"]).first()
    if not user or not bcrypt.checkpw(data["password"].encode(), user.password.encode()):
        return jsonify({"error": "Invalid credentials"}), 401
    token = jwt.encode({
        "user_id": user.id,
        "role": user.role,
        "exp": datetime.utcnow() + timedelta(hours=int(os.getenv("JWT_EXPIRY_HOURS", 24)))
    }, os.getenv("SECRET_KEY"), algorithm="HS256")
    return jsonify({"token": token, "user": user.to_dict()})
```

### Events Routes `/api/events`
```python
# app/routes/events.py
from flask import Blueprint, request, jsonify
from app import db
from app.models.event import Event
from app.middleware.auth_middleware import token_required, role_required

events_bp = Blueprint("events", __name__)

@events_bp.route("", methods=["GET"])
def get_events():
    events = Event.query.all()
    return jsonify([e.to_dict() for e in events])

@events_bp.route("/<int:id>", methods=["GET"])
def get_event(id):
    event = Event.query.get_or_404(id)
    return jsonify(event.to_dict())

@events_bp.route("", methods=["POST"])
@role_required("organizer", "admin")
def create_event(current_user):
    data = request.get_json()
    from datetime import datetime as dt
    event = Event(
        title=data["title"], description=data.get("description"),
        category=data.get("category"), location=data.get("location"),
        date=dt.fromisoformat(data["date"]),
        total_seats=data.get("total_seats", 100),
        price=data.get("price", 0), image_url=data.get("image_url"),
        organizer_id=current_user.id
    )
    db.session.add(event)
    db.session.commit()
    return jsonify(event.to_dict()), 201

@events_bp.route("/<int:id>", methods=["PUT"])
@role_required("organizer", "admin")
def update_event(current_user, id):
    event = Event.query.get_or_404(id)
    if event.organizer_id != current_user.id and current_user.role != "admin":
        return jsonify({"error": "Unauthorized"}), 403
    data = request.get_json()
    for key, val in data.items():
        if hasattr(event, key):
            setattr(event, key, val)
    db.session.commit()
    return jsonify(event.to_dict())

@events_bp.route("/<int:id>", methods=["DELETE"])
@role_required("organizer", "admin")
def delete_event(current_user, id):
    event = Event.query.get_or_404(id)
    if event.organizer_id != current_user.id and current_user.role != "admin":
        return jsonify({"error": "Unauthorized"}), 403
    db.session.delete(event)
    db.session.commit()
    return jsonify({"message": "Event deleted"})
```

### Bookings Routes `/api/bookings`
```python
# app/routes/bookings.py
from flask import Blueprint, request, jsonify
from app import db
from app.models.booking import Booking
from app.models.event   import Event
from app.middleware.auth_middleware import token_required

bookings_bp = Blueprint("bookings", __name__)

@bookings_bp.route("", methods=["POST"])
@token_required
def book_event(current_user):
    data    = request.get_json()
    event   = Event.query.get_or_404(data["event_id"])
    already = Booking.query.filter_by(
        user_id=current_user.id, event_id=event.id, status="confirmed").first()
    if already:
        return jsonify({"error": "Already booked"}), 409
    if event.available_seats <= 0:
        return jsonify({"error": "No seats available"}), 400
    booking = Booking(user_id=current_user.id, event_id=event.id)
    event.booked_seats += 1
    db.session.add(booking)
    db.session.commit()
    return jsonify(booking.to_dict()), 201

@bookings_bp.route("/me", methods=["GET"])
@token_required
def my_bookings(current_user):
    bookings = Booking.query.filter_by(user_id=current_user.id).all()
    return jsonify([b.to_dict() for b in bookings])

@bookings_bp.route("/<int:id>", methods=["DELETE"])
@token_required
def cancel_booking(current_user, id):
    booking = Booking.query.get_or_404(id)
    if booking.user_id != current_user.id:
        return jsonify({"error": "Unauthorized"}), 403
    booking.status = "cancelled"
    booking.event.booked_seats -= 1
    db.session.commit()
    return jsonify({"message": "Booking cancelled"})
```
